Android Gallery 自动无限循环效果

(2012-05-31 16:34:51)
转载▼
标签：

无限

效果图

间隙

线程

定时器

android

gallery

it

 分类： Android
Gallery无限循环，肯定需要模拟用户的动作！，要实现无限循环必须使用handler和线程，效果图和带循环的Gallery效果图差不多
项目结构：

####################################################################################3
main.xml:
<?xml version="1.0" encoding="utf-8"?>
<LinearLayout xmlns:android="http://schemas.android.com/apk/res/android"
    android:orientation="vertical"
    android:layout_width="fill_parent"
    android:layout_height="fill_parent"
    >

<Gallery
         android:id="@+id/gallery"
         android:layout_width="fill_parent"
         android:layout_height="wrap_content">
</Gallery>
</LinearLayout>
#########################################################################################
Activity:
package zkyc.main;


import java.util.Timer;
import java.util.TimerTask;

import android.app.Activity;
import android.content.Context;
import android.os.Bundle;
import android.os.Handler;
import android.os.Message;
import android.view.View;
import android.view.ViewGroup;
import android.widget.BaseAdapter;
import android.widget.Gallery;
import android.widget.ImageView;

public class Study_gallery03Activity extends Activity {
    Gallery gallery;
    public static final int CHANGE_BANNER = 1;
    Handler handler=new Handler(){

        @Override
        public void handleMessage(Message msg) {
            switch(msg.what){

            case 1:

                gallery.setSelection(gallery.getSelectedItemPosition() + 1);
                break;

            default:

            break;

            }
        }
       
       
    };
    @Override
    public void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.main);
        gallery=(Gallery)findViewById(R.id.gallery);
        //设置图片的间隙
        gallery.setSpacing(3);
        ImageAdapter adapter=new ImageAdapter(this);
        gallery.setAdapter( adapter );
        //设置gallery靠左没有间隙（这句话一定要放在setAdapter( adapter )之后，放在前面没有效果）
        gallery.setSelection(adapter.getCount()/2);
        init();
    }
    private  void init(){
        // 这里实现了2中方式其实类似用了java的定时器和线程，可以任意实现一个
//        final Timer timer = new Timer(); 
//       
//        timer.schedule(new TimerTask() { 
//            @Override 
//            public void run() { 
//                handler.sendEmptyMessage(1); 
//            } 
//        }, 1000, 1000);
       
        Thread t=new Thread(new Runnable() {
           
            @Override
            public void run() {
                while(true){
                    handler.sendEmptyMessage(1);
                    try {
                        Thread.sleep(2000);
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
               
            }
        });
        t.start();
       
    }
    class ImageAdapter extends BaseAdapter{
        private Context context;
        //放图片资源
         private Integer[] mImage = {
                 R.drawable.p1,
                 R.drawable.p2,
                 R.drawable.p3,
                 R.drawable.p4,
                 R.drawable.p5,
         };
        public ImageAdapter(Context c ){
            this.context=c;
        }
        @Override
        public int getCount() {
            //定义一个int类型的最大值
            return Integer.MAX_VALUE;
        }

        @Override
        public Object getItem(int position) {
            return position;
        }

        @Override
        public long getItemId(int position) {
            return position;
        }

        @Override
        public View getView(int position, View convertView, ViewGroup parent) {
            ImageView i=new ImageView(context);
            //用postion对图片的个数除模模拟原来图片的个数达到循环
            i.setImageResource(mImage[position%mImage.length]);
            //重新设定图片的宽高
            i.setScaleType(ImageView.ScaleType.FIT_XY);
            //设置ImageView对象的宽高
            i.setLayoutParams(new Gallery.LayoutParams(65, 80));
            return i;
        }
       
       
       
    }
   
}
